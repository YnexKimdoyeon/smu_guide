================================================================================
[9] Nginx 설정
================================================================================

# Nginx 설정 파일 생성
sudo nano /etc/nginx/sites-available/sunmoon-guide

# 아래 내용 붙여넣기:
--------------------------------------------------------------------------------
server {
    listen 80;
    server_name ynex2.mycafe24.com;
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name ynex2.mycafe24.com;

    # SSL 인증서 (certbot이 자동으로 설정함)
    ssl_certificate /etc/letsencrypt/live/ynex2.mycafe24.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/ynex2.mycafe24.com/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    # 보안 헤더
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;

    # 프론트엔드 (Next.js)
    location / {
        proxy_pass http://127.0.0.1:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }

    # 백엔드 API
    location /api {
        proxy_pass http://127.0.0.1:8000/api;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # WebSocket
    location /ws {
        proxy_pass http://127.0.0.1:8000/ws;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_read_timeout 86400;
    }

    # API 문서 (개발 시에만, 프로덕션에서는 제거)
    # location /docs {
    #     proxy_pass http://127.0.0.1:8000/docs;
    # }
}
--------------------------------------------------------------------------------

# 설정 파일 활성화
sudo ln -s /etc/nginx/sites-available/sunmoon-guide /etc/nginx/sites-enabled/

# 기본 설정 제거
sudo rm /etc/nginx/sites-enabled/default

# 설정 테스트
sudo nginx -t

# Nginx 재시작
sudo systemctl restart nginx


================================================================================
[10] PM2로 앱 실행 (자동 재시작)
================================================================================

# 백엔드 실행 스크립트 생성
cat > /var/www/sunmoon-guide/backend/start.sh << 'EOF'
#!/bin/bash
cd /var/www/sunmoon-guide/backend
source venv/bin/activate
exec uvicorn app.main:app --host 127.0.0.1 --port 8000
EOF

chmod +x /var/www/sunmoon-guide/backend/start.sh

# PM2 ecosystem 설정 파일 생성
cat > /var/www/sunmoon-guide/ecosystem.config.js << 'EOF'
module.exports = {
  apps: [
    {
      name: 'sunmoon-frontend',
      cwd: '/var/www/sunmoon-guide',
      script: 'npm',
      args: 'start',
      env: {
        NODE_ENV: 'production',
        PORT: 3000
      }
    },
    {
      name: 'sunmoon-backend',
      cwd: '/var/www/sunmoon-guide/backend',
      script: '/var/www/sunmoon-guide/backend/start.sh',
      interpreter: '/bin/bash'
    }
  ]
}
EOF

# PM2로 앱 시작
cd /var/www/sunmoon-guide
pm2 start ecosystem.config.js

# PM2 상태 확인
pm2 status

# 로그 확인
pm2 logs

# 서버 재부팅 시 자동 시작 설정
pm2 save
pm2 startup
# 출력되는 명령어를 복사해서 실행

================================================================================
[13] 문제 해결
================================================================================

### 502 Bad Gateway
- PM2가 실행 중인지 확인: pm2 status
- 백엔드 로그 확인: pm2 logs sunmoon-backend
- 포트 확인: sudo netstat -tlnp | grep -E '3000|8000'

### WebSocket 연결 실패
- Nginx 설정에서 /ws 경로 확인
- wss:// 프로토콜 사용 확인

### SSL 인증서 오류
- 도메인이 서버 IP를 가리키는지 확인
- certbot 재실행: sudo certbot --nginx -d ynex2.mycafe24.com

### 메모리 부족
- PM2 메모리 확인: pm2 monit
- 불필요한 프로세스 종료
